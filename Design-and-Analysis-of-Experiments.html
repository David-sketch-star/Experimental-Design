2.Analysis of Randomised Block Designs

#create the data
Values <- c(24.7,20.6,27.7,16.2,16.2,24.9,27.3,28.8,22.7,15.0,17.0,22.5,38.5,39.5,36.8,19.6,15.4,26.3,28.5,31.0,34.9,14.1,17.7,22.6)

Blocks <- rep(c("B1","B2","B3","B4"), each = 6)

Treatments_data <- data.frame(Values, Treatments, Blocks)

#Asses the effect of material type and temperature on battery life
#creating the data
Values <- c(130,74,155,180,34,80,40,75,20,82,70,58,150,159,188,126,136
            ,106,122,115,25,58,70,45,138,168,110,160,174,150,120,139,
